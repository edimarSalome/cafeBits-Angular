Injeção de depêndencia

app.js

                            
//Declaração da Controller
app.controller('boasVindasController',['$scope', function($scope){ //Esta controller utiliza $scope
    
    //Definindo um valor para o modelo nome
    this.nome = 'Edimar';
    
    //Função criada em $scope poderá ser acessada pelo DOM da página
    $scope.digaOi = function(){
        alert("Oi "+this.ctrl.nome+" !!!");
    };
}]);

 

dependencias.html

                            
<!--- iniciando a controller através da diretiva ng-controller -->
<div class="row" ng-controller="boasVindasController as ctrl">
    <h2> ctrl.nome </h2>
    <div class="col-sm-3">
        <!--- Ng Model declara o modelo --->
        <select ng-model="ctrl.nome" class="form-control">
            <option>World</option>
            <option>Lifesys</option>
            <option>Calouros</option>
        </select>
    </div>
    <!--- Chama a função de Scope -->
    <div class="col-sm-1">
        <button class="btn btn-primary" ng-click="digaOi()">Diga oi</button>
    </div>
</div>
