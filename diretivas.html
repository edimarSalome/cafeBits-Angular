app.js

                            
//Definindo a Diretiva
app.directive('diretivaProduto', function(){ //A letra maiúscula tem um motivo
    return {
        restrict: 'E', //Será criada como elemento HTML
        templateUrl: 'produto.html', //Arquivo html comum
        controller: function(){ //Define a controller da diretiva
            var produto = {
                nome:'Autolac',
                preço: '1200,00',
                descrição: 'Excelente para Laboratórios de Análises Clínicas.'
            };
        },
        controllerAs: 'ctrl' //Define um alias para a controller
    };
});
